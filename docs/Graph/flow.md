        while(!q.empty())q.pop();

        {
            int k=q.front();q.pop();

            {
                if(v[i]&&!d[to[i]])
                {
                    d[to[i]]=d[k]+1,now[to[i]]=head[to[i]],q.e(to[i]);
                    if(to[i]==T)return 1;
                }
            }
        }
        return 0;
    }
    int dinic(int k,int flow)
    {
        if(k==T)return flow;
        int rest=flow,t;
        for(int i=now[k];i&&rest;i=nex[i])
        {
            now[k]=i;
            if(d[to[i]]!=d[k]+1||!v[i])continue;
            t=dinic(to[i],min(v[i],rest));
            if(!t)d[to[i]]=0;
            rest-=t,v[i]-=t,v[i^1]+=t;
        }
        return flow-rest;
    }
