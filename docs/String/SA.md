    inline int ask(int x,int y)
    {
        if(x==y)return inf;
        if((x=rk[x])>(y=rk[y]))swap(x,y);
        int k=__lg(y-x++);
        return min(F[k][x],F[k][y-(1<<k)+1]);
    }

    {

    }
